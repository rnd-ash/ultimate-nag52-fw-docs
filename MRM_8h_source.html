198 bool import_frames(uint64_t value, uint32_t can_id, uint64_t timestamp_now) {
199 switch(can_id) {
200 case LRW_236_CAN_ID:
201 LAST_FRAME_TIMES[0] = timestamp_now;
202 FRAME_DATA[0] = value;
203 return true;
204 case MRM_238_CAN_ID:
205 LAST_FRAME_TIMES[1] = timestamp_now;
206 FRAME_DATA[1] = value;
207 return true;
208 default:
209 return false;
210 }
211 }
212
220 bool get_LRW_236(uint64_t now, uint64_t max_expire_time, LRW_236* dest) const {
221 if (LAST_FRAME_TIMES[0] == 0 || dest == nullptr) { // CAN Frame has not been seen on bus yet / NULL pointer
222 return false;
223 } else if (now > LAST_FRAME_TIMES[0] && now - LAST_FRAME_TIMES[0] > max_expire_time) { // CAN Frame has not refreshed in valid interval
224 return false;
225 } else { // CAN Frame is valid! return it
226 dest->raw = FRAME_DATA[0];
227 return true;
228 }
229 }
230
238 bool get_MRM_238(uint64_t now, uint64_t max_expire_time, MRM_238* dest) const {
239 if (LAST_FRAME_TIMES[1] == 0 || dest == nullptr) { // CAN Frame has not been seen on bus yet / NULL pointer
240 return false;
241 } else if (now > LAST_FRAME_TIMES[1] && now - LAST_FRAME_TIMES[1] > max_expire_time) { // CAN Frame has not refreshed in valid interval
242 return false;
243 } else { // CAN Frame is valid! return it
244 dest->raw = FRAME_DATA[1];
245 return true;
246 }
247 }
