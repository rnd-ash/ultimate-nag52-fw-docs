278 bool import_frames(uint64_t value, uint32_t can_id, uint64_t timestamp_now) {
279 switch(can_id) {
280 case KOMBI_408_CAN_ID:
281 LAST_FRAME_TIMES[0] = timestamp_now;
282 FRAME_DATA[0] = value;
283 return true;
284 case KOMBI_412_CAN_ID:
285 LAST_FRAME_TIMES[1] = timestamp_now;
286 FRAME_DATA[1] = value;
287 return true;
288 default:
289 return false;
290 }
291 }
292
300 bool get_KOMBI_408(uint64_t now, uint64_t max_expire_time, KOMBI_408* dest) const {
301 if (LAST_FRAME_TIMES[0] == 0 || dest == nullptr) { // CAN Frame has not been seen on bus yet / NULL pointer
302 return false;
303 } else if (now > LAST_FRAME_TIMES[0] && now - LAST_FRAME_TIMES[0] > max_expire_time) { // CAN Frame has not refreshed in valid interval
304 return false;
305 } else { // CAN Frame is valid! return it
306 dest->raw = FRAME_DATA[0];
307 return true;
308 }
309 }
310
318 bool get_KOMBI_412(uint64_t now, uint64_t max_expire_time, KOMBI_412* dest) const {
319 if (LAST_FRAME_TIMES[1] == 0 || dest == nullptr) { // CAN Frame has not been seen on bus yet / NULL pointer
320 return false;
321 } else if (now > LAST_FRAME_TIMES[1] && now - LAST_FRAME_TIMES[1] > max_expire_time) { // CAN Frame has not refreshed in valid interval
322 return false;
323 } else { // CAN Frame is valid! return it
324 dest->raw = FRAME_DATA[1];
325 return true;
326 }
327 }

329 private:

331 uint64_t LAST_FRAME_TIMES[2];
