193 switch(can_id) {
194 case SBW_RS_ISM_CAN_ID:
195 LAST_FRAME_TIMES[0] = timestamp_now;
196 FRAME_DATA[0] = value;
197 return true;
198 case NM_TSLM_CAN_ID:
199 LAST_FRAME_TIMES[1] = timestamp_now;
200 FRAME_DATA[1] = value;
201 return true;
202 default:
203 return false;
204 }
205 }

214 bool get_SBW_RS_ISM(uint64_t now, uint64_t max_expire_time, SBW_RS_ISM* dest) const {
215 if (LAST_FRAME_TIMES[0] == 0 || dest == nullptr) { // CAN Frame has not been seen on bus yet / NULL pointer
216 return false;
217 } else if (now > LAST_FRAME_TIMES[0] && now - LAST_FRAME_TIMES[0] > max_expire_time) { // CAN Frame has not refreshed in valid interval
218 return false;
219 } else { // CAN Frame is valid! return it
220 dest->raw = FRAME_DATA[0];
221 return true;
222 }
223 }
