12template<typename T, uint8_t MAX_SIZE> struct MovingAverage {
13 T readings[MAX_SIZE];
14 uint8_t sample_id;
15 uint64_t sum;
16
17 MovingAverage(void) {
18 this->sample_id = 0;
19 this->sum = 0;
20 memset(this->readings, 0x00, sizeof(this->readings));
21 }
22
23 void add_to_sample(T reading) {
24 this->sum -= this->readings[this->sample_id];
25 this->readings[this->sample_id] = reading;
26 this->sum += reading;
27 this->sample_id = (this->sample_id+1) % MAX_SIZE;
28 }
29
30 T get_average(void) {
31 return (float)sum / (float)MAX_SIZE;
32 }
33};
