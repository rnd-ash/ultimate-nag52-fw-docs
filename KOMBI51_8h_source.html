173 bool import_frames(uint64_t value, uint32_t can_id, uint64_t timestamp_now) {
174 switch(can_id) {
175 case KOMBI_408EGS51_CAN_ID:
176 LAST_FRAME_TIMES[0] = timestamp_now;
177 FRAME_DATA[0] = value;
178 return true;

180 return false;

191 bool get_KOMBI_408(uint64_t now, uint64_t max_expire_time, KOMBI_408EGS51* dest) const {
192 if (LAST_FRAME_TIMES[0] == 0 || dest == nullptr) { // CAN Frame has not been seen on bus yet / NULL pointer
193 return false;
194 } else if (now > LAST_FRAME_TIMES[0] && now - LAST_FRAME_TIMES[0] > max_expire_time) { // CAN Frame has not refreshed in valid interval
195 return false;
196 } else { // CAN Frame is valid! return it
197 dest->raw = FRAME_DATA[0];
198 return true;
199 }
200 }

204 uint64_t LAST_FRAME_TIMES[1];
