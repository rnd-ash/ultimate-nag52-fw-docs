343 public:
351 bool import_frames(uint64_t value, uint32_t can_id, uint64_t timestamp_now) {
352 switch(can_id) {
353 case BS_200EGS51_CAN_ID:
354 LAST_FRAME_TIMES[0] = timestamp_now;
355 FRAME_DATA[0] = value;
356 return true;
357 case BS_208EGS51_CAN_ID:
358 LAST_FRAME_TIMES[1] = timestamp_now;
359 FRAME_DATA[1] = value;
360 return true;
361 default:
362 return false;
363 }
364 }
365
373 bool get_BS_200(uint64_t now, uint64_t max_expire_time, BS_200EGS51* dest) const {
374 if (LAST_FRAME_TIMES[0] == 0 || dest == nullptr) { // CAN Frame has not been seen on bus yet / NULL pointer
375 return false;
376 } else if (now > LAST_FRAME_TIMES[0] && now - LAST_FRAME_TIMES[0] > max_expire_time) { // CAN Frame has not refreshed in valid interval
377 return false;
378 } else { // CAN Frame is valid! return it
379 dest->raw = FRAME_DATA[0];
380 return true;
381 }
382 }
383
391 bool get_BS_208(uint64_t now, uint64_t max_expire_time, BS_208EGS51* dest) const {
392 if (LAST_FRAME_TIMES[1] == 0 || dest == nullptr) { // CAN Frame has not been seen on bus yet / NULL pointer
393 return false;
394 } else if (now > LAST_FRAME_TIMES[1] && now - LAST_FRAME_TIMES[1] > max_expire_time) { // CAN Frame has not refreshed in valid interval
395 return false;
396 } else { // CAN Frame is valid! return it
397 dest->raw = FRAME_DATA[1];
398 return true;
399 }
400 }
401
402 private:
403 uint64_t FRAME_DATA[2];
404 uint64_t LAST_FRAME_TIMES[2];
405};
