245 bool import_frames(uint64_t value, uint32_t can_id, uint64_t timestamp_now) {
246 switch(can_id) {
247 case FSCM_STAT_CAN_ID:
248 LAST_FRAME_TIMES[0] = timestamp_now;
249 FRAME_DATA[0] = value;
250 return true;
251 case NM_FSCM_CAN_ID:
252 LAST_FRAME_TIMES[1] = timestamp_now;
253 FRAME_DATA[1] = value;
254 return true;
255 default:
256 return false;
257 }
258 }
259
267 bool get_FSCM_STAT(uint64_t now, uint64_t max_expire_time, FSCM_STAT* dest) const {
268 if (LAST_FRAME_TIMES[0] == 0 || dest == nullptr) { // CAN Frame has not been seen on bus yet / NULL pointer
269 return false;
270 } else if (now > LAST_FRAME_TIMES[0] && now - LAST_FRAME_TIMES[0] > max_expire_time) { // CAN Frame has not refreshed in valid interval
271 return false;
272 } else { // CAN Frame is valid! return it
273 dest->raw = FRAME_DATA[0];
274 return true;
275 }
276 }

285 bool get_NM_FSCM(uint64_t now, uint64_t max_expire_time, NM_FSCM* dest) const {

289 return false;
290 } else { // CAN Frame is valid! return it
291 dest->raw = FRAME_DATA[1];
292 return true;
293 }
294 }
